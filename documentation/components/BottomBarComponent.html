import { Component, EventEmitter, Input, Output } from '@angular/core';
import { Router } from '@angular/router';
import { ThemeService } from 'src/app/services/theme.service';

export interface IBottomBarOption {
  icon: string;
  target?: string;
  routerLink?: string;
}

@Component({
  selector: 'app-bottom-bar',
  templateUrl: './bottom-bar.component.html',
  styleUrls: ['./bottom-bar.component.scss']
})
export class BottomBarComponent {
  @Output() optionClick: EventEmitter<any> = new EventEmitter<any>();
  @Output() actionClick: EventEmitter<any> = new EventEmitter<any>();

  @Input() mainOption: IBottomBarOption = {
    icon: 'bookmark_add',
    target: 'addPassage',
    routerLink: '',
  }
  @Input() leftOptions: IBottomBarOption[] = [
    {
      icon: 'library_books',
      target: 'libraryBooksTarget',
      routerLink: '',
    },
    {
      icon: 'bookmarks',
      target: 'passagesTarget',
      routerLink: '',
    }
  ];
  @Input() rightOptions: IBottomBarOption[] = [
    {
      icon: 'category',
      target: 'manageScoringTarget',
      routerLink: '',
    },
    {
      icon: 'auto_graph',
      target: 'insightsTarget',
      routerLink: '',
    }
  ];

  constructor(
    private router: Router,
    public theme: ThemeService
  ) { }


  onOptionClick(value: any) { this.optionClick.emit(value); }
  onRouteActivate(route: string) { this.router.navigateByUrl(route); }
  onActionClick(value: any) { this.actionClick.emit(value); }
}

<div [ngClass]="(theme.isDark$ | async) ? 'bottom-bar-wrapper is-dark' : 'bottom-bar-wrapper'">
  <div class="bar-content">
    <div class="left-content">
      <a
        (click)="(!option.routerLink) ? onOptionClick(option.target) : onRouteActivate(option.routerLink)"
        *ngFor="let option of leftOptions; let i = index;"
        [ngClass]="!option.routerLink && !option.target ? 'option placeholder' : 'option'">
        <i class="material-icons">{{ option.routerLink || option.target ? option.icon : '' }}</i>
      </a>
    </div>

    <a
      (click)="mainOption.routerLink ? onActionClick(mainOption.routerLink) : onActionClick(mainOption.target)"
      class="action-button">
      <i class="material-icons">
        {{ mainOption.icon }}
      </i>
    </a>

    <div class="right-content">
      <a
        (click)="(!option.routerLink) ? onOptionClick(option.target) : onRouteActivate(option.routerLink)"
        [ngClass]="!option.routerLink && !option.target ? 'option placeholder' : 'option'"
        *ngFor="let option of rightOptions; let i = index;">
        <i class="material-icons">{{ option.routerLink || option.target ? option.icon : '' }}</i>
      </a>
    </div>
  </div>
</div>
